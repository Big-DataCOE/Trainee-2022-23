#include<iostream>
using namespace std;

int main(){
    int n;
    cin>>n;
    int x=0;
    while(n--){
        string e;
        cin>>e;
          
        for(int i=0;i<3;i++){
            if(e[i]=='+'){
            x++;
            break;
            }
            else if(e[i]=='-'){
            x--;
            break;
            }
        }
        
        
    }
    cout<<x<<endl;
    return 0;
}
